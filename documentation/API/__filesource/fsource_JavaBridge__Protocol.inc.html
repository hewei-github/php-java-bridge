  299.   function parseHeaders() {
  300.     $this->headers = array();
  301.     while (($str = trim(fgets($this->socket, JAVA_RECV_SIZE)))) {
  302.       if($str[0]=='X') {
  303.         if(!strncasecmp("X_JAVABRIDGE_REDIRECT", $str, 21)) {
  304.           $this->headers["redirect"]=trim(substr($str, 22));
  305.         } else if(!strncasecmp("X_JAVABRIDGE_CONTEXT", $str, 20)) {
  306.           $this->headers["context"]=trim(substr($str, 21));
  307.         }
  308.       } else if($str[0]=='S') {    // Set-Cookie:
  309.         if(!strncasecmp("SET-COOKIE", $str, 10)) {
  310.           $str=substr($str, 12);
  311.           $ar = explode(";", $str);
  312.           $cookie = explode("=",$ar[0]);
  313.           $path = "";
  314.           if(isset($ar[1])) $p=explode("=", $ar[1]);
  315.           if(isset($p)) $path=$p[1];
  316.           $this->doSetCookie($cookie[0], $cookie[1], $path);
  317.         }
  318.       } else if($str[0]=='C') { // Content-Length
  319.         if(!strncasecmp("CONTENT-LENGTH", $str, 14)) {
  320.           $this->headers["content_length"]=trim(substr($str, 15));
  321.         } else if(!strncasecmp("CONNECTION", $str, 10)) {
  322.           $this->headers["connection"]=trim(substr($str, 11));
  323.         }
  324.       }
  325.     }
  326.   }

  379.   function write($data) {
  380.  
  381.     $compatibility = java_getCompatibilityOption($this->protocol->client);
  382.     $this->headers = null;
  383.     $socket = $this->socket;
  384.     $len = 2 + strlen($data);
  385.     $webapp = $this->getWebApp();
  386.     $cookies = $this->getCookies();
  387.     $channel = $this->getChannelName();
  388.     $context = $this->getContext();
  389.     $redirect = $this->redirect;
  390.     $res = "PUT ";
  391.     $res .= $webapp;
  392.     $res .= JAVA_PERSISTENT_SERVLET_CONNECTIONS ? " HTTP/1.1\r\n" : " HTTP/1.0\r\n"; 
  393.     $res .= "Host: {$this->host}:{$this->port}\r\n";
  394.     $res .= "Content-Length: "; $res .= $len; $res .= "\r\n";
  395.     $res .= $context;
  396.     $res .= $cookies;
  397.     $res .= $redirect;
  398.     if(!is_null($channel)) $res .= $channel;
  399.     $res .= "\r\n";
  400.     $res .= "\177";
  401.     $res .= $compatibility;
  402.     $res .= $data;
  403.     
  404.     $count = fwrite($socket, $res) or $this->shutdownBrokenConnection("Broken connection handle");
  405.     fflush($socket)                or $this->shutdownBrokenConnection("Broken connection handle");
  406.     
  407.     return $count;
  408.   }
